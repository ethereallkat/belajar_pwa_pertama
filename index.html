import '../../features/scan/domain/entities/skin_profile.dart'; import '../../features/tracker/domain/entities/skin_log.dart'; class GlowScoreCalculator { /// Computes a 0–100 glow score from multiple skin parameters. static GlowScore calculate({ required SkinProfile profile, required List<SkinLog> recentLogs, required int streakDays, }) { final hydrationScore = _scoreHydration(profile.hydrationLevel); final acneScore = _scoreAcne(profile.acneLevel); final oilBalanceScore = _scoreOilBalance(profile.oilinessLevel); final textureScore = _scoreTexture(profile.textureRating); final consistencyScore = _scoreConsistency(recentLogs); final streakBonus = (streakDays * 0.3).clamp(0.0, 10.0); final raw = ( hydrationScore * 0.25 + acneScore * 0.20 + oilBalanceScore * 0.15 + textureScore * 0.20 + consistencyScore * 0.20 ) * 100 + streakBonus; final total = raw.clamp(0.0, 100.0).round(); return GlowScore( total: total, breakdown: { 'hydration': hydrationScore, 'acne': acneScore, 'oil_balance': oilBalanceScore, 'texture': textureScore, 'consistency': consistencyScore, }, tier: _scoreToTier(total), advice: _generateAdvice(total, profile), ); } static double _scoreHydration(double level) => (level / 100.0).clamp(0.0, 1.0); static double _scoreAcne(AcneLevel level) => switch (level) { AcneLevel.none => 1.0, AcneLevel.mild => 0.75, AcneLevel.moderate => 0.45, AcneLevel.severe => 0.20, }; static GlowTier _scoreToTier(int score) => switch (score) { >= 85 => GlowTier.radiant, >= 70 => GlowTier.glowing, >= 50 => GlowTier.healthy, _ => GlowTier.building, }; }

class RecommendationEngine { final SkincareRepository _skincareRepo; final MakeupRepository _makeupRepo; RecommendationEngine(this._skincareRepo, this._makeupRepo); Future<BeautyRecommendation> recommend(ScanResult scan) async { final [skincare, makeup] = await Future.wait([ _buildSkincareRoutine(scan.skinProfile), _buildMakeupRecommendation(scan), ]); return BeautyRecommendation(skincare: skincare, makeup: makeup); } Future<SkincareRoutine> _buildSkincareRoutine(SkinProfile profile) async { final allProducts = await _skincareRepo.getAllProducts(); final filtered = allProducts.where((p) => p.skinTypeCompatibility.contains(profile.skinType) && p.suitableConcerns.any((c) => profile.concerns.contains(c)) ).toList(); final amRoutine = filtered .where((p) => p.usageTime != UsageTime.pm) .toList() ..sort((a, b) => a.routineStep.compareTo(b.routineStep)); final pmRoutine = filtered .where((p) => p.usageTime != UsageTime.am) .toList() ..sort((a, b) => a.routineStep.compareTo(b.routineStep)); return SkincareRoutine(am: amRoutine, pm: pmRoutine); } Future<MakeupRecommendation> _buildMakeupRecommendation(ScanResult scan) async { final all = await _makeupRepo.getAllProducts(); return MakeupRecommendation( foundation: _matchFoundation(all, scan), lipColor: _matchLip(all, scan), blush: _matchBlush(all, scan), eyeshadow: _matchEyeshadow(all, scan), ); } SkincareProduct _matchFoundation(List<MakeupProduct> all, ScanResult s) => all.firstWhere((p) => p.category == MakeupCategory.foundation && p.undertoneMatch.contains(s.skinProfile.undertone) && (s.skinProfile.oilinessLevel > 0.5 ? p.finishType == FinishType.matte : p.finishType == FinishType.dewy), ); }
